* profiler
#+begin_src emacs-lisp :tangle no
  (require 'profiler)
  (profiler-start 'cpu)
  (add-hook 'emacs-startup-hook
            (lambda ()
              (message "Emacs initialized in %.3f sec with %d garbage collection(s)"
                       (float-time (time-subtract after-init-time before-init-time))
                       gcs-done)))
#+end_src

* performance tuning for setopt
#+begin_src emacs-lisp
  ;; Temporarily disabe `setopt' because it is slower than `setq'
  (fset 'real-setopt (symbol-function 'setopt))
  (defmacro setopt (&rest args)
    "setopt を setq に一時的に変えるためのマクロ."
    `(setq ,@args))
  (add-hook 'emacs-startup-hook
            (lambda ()
              (fset 'setopt (symbol-function 'real-setopt))
              (fmakunbound 'real-setopt)
              ))
#+end_src
